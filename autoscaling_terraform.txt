 [root@Terraform autoscaling]# cat main.tf 
# Provider information

provider "aws" {
region = "us-west-2"
profile = "default"
}

# Gathering Availability zone information

data "aws_availability_zones" "all" {}


# Creating My Security Group

resource "aws_security_group" "allow_all" {
  name        = "allow_all"
  description = "Allow all inbound traffic"
 

  ingress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port       = 0
    to_port         = 0
    protocol        = "-1"
    cidr_blocks     = ["0.0.0.0/0"]
    
  }
}

# Creating loadbalencer

# Create a new load balancer
resource "aws_elb" "myelb" {
  name               = "myelb"
  availability_zones = ["${data.aws_availability_zones.all.names}"]

  listener {
    instance_port     = 80
    instance_protocol = "http"
    lb_port           = 80
    lb_protocol       = "http"
  }


  health_check {
    healthy_threshold   = 2
    unhealthy_threshold = 2
    timeout             = 3
    target              = "TCP:80"
    interval            = 30
  }

  tags {
    Name = "dvsbatch8_elb"
  }
}

# Creating launch configuration 

resource "aws_launch_configuration" "my_lcf" {
  image_id      = "ami-a0cfeed8"
  instance_type = "t2.micro"
  security_groups = ["${aws_security_group.allow_all.id}"]
  user_data = <<-EOF
            #!/bin/bash
            yum install httpd -y
            chkconfig httpd on
            echo "Hello, World" >> /var/www/html/index.html
            service httpd restart
            EOF

}

# Scalling policeis

resource "aws_autoscaling_group" "myautoscaling" {
  name                      = "myautoscaling"
  max_size                  = 10
  min_size                  = 2
  health_check_grace_period = 300
  health_check_type         = "ELB"
  desired_capacity          = 2
  force_delete              = true
  launch_configuration      = "${aws_launch_configuration.my_lcf.name}"
  load_balancers = ["${aws_elb.myelb.id}"]
  availability_zones = ["${data.aws_availability_zones.all.names}"]
  tag {
    key                 = "Name"
    value               = "MyAutoServers"
    propagate_at_launch = true
  }
 }

# Printing LB endpoint

output "loadbalencer" {
value = "${aws_elb.myelb.dns_name}"
}
